const responseJson=pm.response.json();
pm.test("Status code is 200", function () {
    pm.response.to.have.status(200);
});

pm.test(" Content-Type is JSON", function () {
    pm.expect(pm.response.headers.get("Content-Type")).to.include("application/json");
});

pm.test("Response contains hierarchies array",function(){
    pm.expect(responseJson).to.have.property('hierarchies').that.is.an('array');
});

pm.test("Each hierarchy object contains hierarchyId and hierarchyName", function () {
    responseJson.hierarchies.forEach(dimension => {
        pm.expect(dimension).to.have.property('hierarchyId');
        pm.expect(dimension).to.have.property('hierarchyName');
    });
});
