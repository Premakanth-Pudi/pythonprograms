const responseJson=pm.response.json();

pm.test("Status code is 500", function () {
    pm.response.to.have.status(500);
});

pm.test("Status response is INTERNAL_SERVER_ERROR", function(){
    pm.expect(responseJson.status).to.eql("INTERNAL_SERVER_ERROR")
});

pm.test("Error Message is: Internal Error", function(){
    pm.expect(responseJson.errors).to.include("Internal Error");
});

pm.test("Path is /sustainability/v1/applications/invalid_applicationID/dimensions", function(){
    pm.expect(responseJson.path).to.eql("/sustainability/v1/applications/invalid_applicationID/dimensions") 
});
